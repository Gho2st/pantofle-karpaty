import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: [
        '/koszyk/',
        '/zamowienie/',
        '/admin/',
        '/moje-konto/'
      ],
    },
    sitemap: 'https://sklep-pantofle-karpaty.pl/sitemap.xml',
  }
}
